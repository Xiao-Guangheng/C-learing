#include <iostream>
using namespace std;

template <typename T>
void bubbleSort(T arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                T temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int a[] = {5, 2, 8, 1, 3};
    bubbleSort(a, 5);
    for (int i = 0; i < 5; i++) cout << a[i] << " ";
    cout << endl;

    char b[] = {'d', 'a', 'c', 'b'};
    bubbleSort(b, 4);
    for (int i = 0; i < 4; i++) cout << b[i] << " ";
    cout << endl;

    return 0;
}
